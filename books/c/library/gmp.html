The GNU Multiple Precision Arithmetic Library

GNU 多精度运算库

主页：https://gmplib.org
手册：https://gmplib.org/manual/

使用示例：

#include <gmp.h>

int main(int argc, char *argv[])
{
    mpz_t       i1, i2, i3;
    mpz_init_set_str(i1, "151542622415", 10);
    mpz_init_set_str(i2, "151542622415", 10);
    mpz_init(i3);

    mpz_mul(i3, i1, i2);

    gmp_printf("%Zd\n", i3);

    return 0;
}
